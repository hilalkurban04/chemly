function checkTF(btn, isCorrectAnswer, id, explanation) { const res = document.getElementById(id); const btns = btn.parentElement.querySelectorAll('button'); // Tüm butonları devre dışı bırak btns.forEach(b => b.disabled = true); const userChoseTrue = btn.textContent.trim() === 'Doğru'; const userCorrect = userChoseTrue === isCorrectAnswer; if (userCorrect) { // Kullanıcı doğru cevapladıysa: Sadece seçtiği şık yeşil yansın btn.classList.add('correct-ans'); res.innerHTML = '✅ ' + explanation; res.style.color = '#22c55e'; } else { // Kullanıcı yanlış cevapladıysa: Sadece seçtiği şık kırmızı yansın btn.classList.add('wrong-ans'); res.innerHTML = '❌ ' + explanation; res.style.color = '#ef4444'; } }
